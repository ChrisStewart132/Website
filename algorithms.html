SELECTION SORT

void selection_sort(int* nums, int numsSize) {
    for(int i = 0; i < numsSize; i++){
        int* ptr = &nums[i];
        for(int j = i+1; j < numsSize; j++){
            if(nums[j] < *ptr){
                ptr = &nums[j];
            }
        }
        int temp = nums[i];
        nums[i] = *ptr;
        *ptr = temp;
    }
}

INSERTION SORT

void insertion_sort(int* nums, int numsSize){
    for(int i = 1; i < numsSize; i++){
        int j = i;
        int current = nums[j];
        while(j > 0 && nums[j-1] > current){
            nums[j] = nums[j-1];
            j--;
        }
        nums[j] = current;
    }
}
